## 一、机械化问题修正方案

### 1.1 高频快照问题

#### ❌ 原错误代码
```javascript
async function waitForPageLoad(timeout = 20000) {
  while (Date.now() - startTime < timeout) {
    snapshot = await browser.snapshot({ refs: 'aria' });  // 每秒一次！
    if (!snapshot.text?.includes('正在加载中')) break;
    await sleep(1000);  // 固定 1 秒间隔
  }
}
```

**问题分析：** 每秒调用一次 `browser.snapshot()` 相当于每秒"敲门"一次，服务端日志会显示异常的 DOM 访问频率。

#### ✅ 修正方案：指数退避 + 事件驱动

```javascript
async function waitForPageLoad(timeout = 30000) {
  const checkIntervals = [3000, 5000, 8000, 10000]; // 递增间隔
  let attempt = 0;
  const startTime = Date.now();
  
  while (Date.now() - startTime < timeout) {
    // 只在关键时间点检查，不是轮询
    const interval = checkIntervals[Math.min(attempt, checkIntervals.length - 1)];
    await sleep(interval);
    attempt++;
    
    // 只在第 2、4 次检查时获取快照（减少 50% 调用）
    if (attempt % 2 === 0) {
      snapshot = await browser.snapshot({ refs: 'aria', timeoutMs: 5000 });
      
      // 检查关键元素是否存在（而不是检查"加载中"文本）
      const hasJobList = snapshot.querySelector('[ref*="job"]') || 
                         snapshot.querySelector('link[href*="/job_detail/"]');
      
      if (hasJobList) {
        console.log('✅ 页面已就绪，检测到职位列表');
        return snapshot;
      }
    }
  }
  
  // 超时后返回 null，由上层决定重试或放弃
  console.warn('⚠️ 页面加载超时，返回 null');
  return null;
}
```

**核心改进：**
- 检查间隔从固定 1 秒改为 3s→5s→8s→10s 递增
- 只在偶数次检查时获取快照（减少 50% DOM 访问）
- 检测**关键元素存在**而非"加载中"文本消失
- 超时返回 null 而非抛异常，避免死循环

### 1.2 单一操作路径问题

#### ❌ 原错误行为
```
打开浏览器 → 直接导航到搜索 URL → 抓取 → 结束
```

**问题分析：** 正常用户不会直接访问深层 URL，而是从首页开始搜索、点击、翻页。

#### ✅ 修正方案：模拟真实用户路径

```javascript
async function simulateRealUserSearch(keyword) {
  // 步骤 1: 访问首页（建立正常访问记录）
  await browser.open('https://www.zhipin.com/');
  await sleep(randomRange(5000, 8000)); // 5-8 秒阅读首页
  
  // 步骤 2: 随机滚动页面（模拟浏览行为）
  await simulateScroll();
  await sleep(randomRange(3000, 5000));
  
  // 步骤 3: 找到搜索框并输入（而不是直接改 URL）
  const searchBox = await findElement('textbox', '搜索职位、公司');
  if (searchBox) {
    await browser.act({ kind: 'type', ref: searchBox.ref, text: keyword, slowly: true });
    await sleep(randomRange(1000, 2000)); // 输入后停顿
    
    // 步骤 4: 点击搜索按钮
    const searchBtn = await findElement('button', '搜索');
    if (searchBtn) {
      // 点击前模拟鼠标移动轨迹
      await simulateMouseMovement(searchBtn);
      await browser.act({ kind: 'click', ref: searchBtn.ref });
    }
  }
  
  // 步骤 5: 等待搜索结果加载
  await waitForPageLoad();
  
  // 步骤 6: 随机滚动浏览结果
  await simulateScroll();
  await sleep(randomRange(4000, 7000));
}

async function simulateScroll() {
  // 随机滚动 2-4 次，每次滚动距离随机
  const scrollTimes = randomRange(2, 4);
  for (let i = 0; i < scrollTimes; i++) {
    const scrollDistance = randomRange(100, 400);
    await browser.evaluate(() => window.scrollBy(0, scrollDistance));
    await sleep(randomRange(800, 2000)); // 每次滚动后停顿
  }
}

async function simulateMouseMovement(targetElement) {
  // 生成贝塞尔曲线路径（模拟人类鼠标轨迹）
  const path = generateBezierPath(targetElement);
  for (const point of path) {
    await browser.evaluate((x, y) => {
      // 模拟鼠标移动（需要注入脚本）
      window._mockMouseX = x;
      window._mockMouseY = y;
    }, point.x, point.y);
    await sleep(randomRange(50, 150)); // 每点 50-150ms
  }
}
```

**核心改进：**
- 从首页开始，完整模拟用户路径
- 搜索框输入而非 URL 跳转
- 随机滚动 + 鼠标轨迹模拟
- 每次操作后有不规则停顿

### 1.3 廉价延迟问题

#### ❌ 原错误代码
```javascript
await randomDelay(3000, 8000); // 以为这样就能瞒天过海
```

**问题分析：** 单纯的延迟没有意义，风控看的是**行为模式**而非时间间隔。

#### ✅ 修正方案：有意义的行为填充

```javascript
async function meaningfulDelay(context) {
  const behaviors = [
    // 行为 1: 阅读页面内容（随机滚动）
    async () => {
      await simulateScroll();
      console.log('📖 模拟阅读页面');
    },
    
    // 行为 2: 查看公司信息（点击公司链接）
    async () => {
      if (Math.random() > 0.6) { // 40% 概率执行
        const companyLink = await findElement('link', '公司');
        if (companyLink) {
          await browser.act({ kind: 'click', ref: companyLink.ref });
          await sleep(randomRange(5000, 10000)); // 浏览公司页 5-10 秒
          await browser.navigate('back'); // 返回
        }
      }
    },
    
    // 行为 3: 切换标签页（模拟多任务）
    async () => {
      if (Math.random() > 0.7) { // 30% 概率执行
        await browser.evaluate(() => document.hidden); // 模拟切出
        await sleep(randomRange(2000, 5000));
      }
    },
    
    // 行为 4: 无操作发呆（人类会走神）
    async () => {
      await sleep(randomRange(3000, 8000));
      console.log('😐 模拟发呆');
    }
  ];
  
  // 随机选择 1-2 个行为执行
  const selectedBehaviors = shuffle(behaviors).slice(0, randomRange(1, 2));
  for (const behavior of selectedBehaviors) {
    await behavior();
    await sleep(randomRange(1000, 3000));
  }
}
```

**核心改进：**
- 延迟期间执行**有意义的行为**
- 行为选择随机化
- 包含"发呆"这种人类特征

## 二、异常处理强化方案

### 2.1 逻辑死循环问题

#### ❌ 原错误逻辑
```javascript
try {
  await click(selector, { timeout: 8000 });
} catch (error) {
  // 重试...重试...重试...（无限循环）
  await click(selector, { timeout: 8000 });
}
```

#### ✅ 修正方案：状态检测 + 熔断机制

```javascript
class CircuitBreaker {
  constructor(options = {}) {
    this.failureThreshold = options.failureThreshold || 3;
    this.resetTimeout = options.resetTimeout || 300000; // 5 分钟
    this.failures = 0;
    this.lastFailureTime = null;
    this.state = 'CLOSED'; // CLOSED, OPEN, HALF_OPEN
  }
  
  async execute(operation) {
    if (this.state === 'OPEN') {
      const timeSinceFailure = Date.now() - this.lastFailureTime;
      if (timeSinceFailure > this.resetTimeout) {
        console.log('🔌 熔断器进入 HALF_OPEN 状态，尝试恢复');
        this.state = 'HALF_OPEN';
      } else {
        throw new Error('熔断器已打开，拒绝执行');
      }
    }
    
    try {
      const result = await operation();
      if (this.state === 'HALF_OPEN') {
        console.log('✅ 操作成功，熔断器恢复 CLOSED');
        this.state = 'CLOSED';
        this.failures = 0;
      }
      return result;
    } catch (error) {
      this.failures++;
      this.lastFailureTime = Date.now();
      
      if (this.failures >= this.failureThreshold) {
        console.error(`🚫 熔断器触发（${this.failures}次失败）`);
        this.state = 'OPEN';
      }
      
      throw error;
    }
  }
}

// 使用示例
const circuitBreaker = new CircuitBreaker({ failureThreshold: 3 });

async function safeClick(selector) {
  return await circuitBreaker.execute(async () => {
    // 先检查页面状态
    const snapshot = await browser.snapshot({ refs: 'aria' });
    
    // 检测验证码
    if (snapshot.text?.includes('验证码') || 
        snapshot.text?.includes('安全验证') ||
        snapshot.text?.includes('滑动验证')) {
      throw new Error('检测到验证码，触发熔断');
    }
    
    // 检测异常页面
    if (snapshot.text?.includes('访问异常') || 
        snapshot.text?.includes('请稍后再试')) {
      throw new Error('检测到异常页面，触发熔断');
    }
    
    // 正常点击
    await browser.act({ kind: 'click', ref: selector });
  });
}
```

**核心改进：**
- 熔断器模式：3 次失败后自动停止
- 执行前检测页面状态（验证码、异常提示）
- 5 分钟冷却期后尝试恢复
- 明确的状态转换（CLOSED→OPEN→HALF_OPEN）

### 2.2 过度清理缓存问题

#### ❌ 原错误行为
```javascript
// 每次搜索前都清理
await browser.clearCookies('zhipin.com');
await browser.clearCache();
```

**问题分析：** 正常用户不会每次访问都清空 Cookie，这相当于每次都装成"新用户"。

#### ✅ 修正方案：持久化会话 + 选择性清理

```javascript
class SessionManager {
  constructor() {
    this.sessionFile = 'boss-session.json';
    this.session = this.loadSession();
  }
  
  loadSession() {
    try {
      if (fs.existsSync(this.sessionFile)) {
        const data = fs.readFileSync(this.sessionFile, 'utf8');
        return JSON.parse(data);
      }
    } catch (e) {
      console.warn('读取会话失败，创建新会话');
    }
    return {
      createdAt: Date.now(),
      lastVisit: Date.now(),
      visitCount: 0,
      cookies: []
    };
  }
  
  saveSession() {
    fs.writeFileSync(this.sessionFile, JSON.stringify(this.session, null, 2));
  }
  
  updateVisit() {
    this.session.lastVisit = Date.now();
    this.session.visitCount++;
    this.saveSession();
  }
  
  // 只在必要时清理（如检测到会话污染）
  shouldClearCookies() {
    // 会话超过 7 天，清理
    if (Date.now() - this.session.createdAt > 7 * 24 * 60 * 60 * 1000) {
      console.log('🗑️ 会话超过 7 天，清理 Cookie');
      return true;
    }
    
    // 访问次数过多（>50 次），清理
    if (this.session.visitCount > 50) {
      console.log('🗑️ 访问次数过多，清理 Cookie');
      return true;
    }
    
    return false;
  }
}

// 使用
const sessionManager = new SessionManager();

async function initSession() {
  if (sessionManager.shouldClearCookies()) {
    await browser.clearCookies('zhipin.com');
    sessionManager = new SessionManager(); // 重置会话
  }
  
  sessionManager.updateVisit();
}
```

**核心改进：**
- 持久化会话文件（记录访问历史）
- 只在必要时清理（7 天或 50 次访问后）
- 建立"老用户"形象

#### ✅ 修正方案：IP 质量检查 + 代理池

```javascript
class IPChecker {
  constructor() {
    this.blacklistedIPs = new Set();
    this.lastCheckTime = 0;
    this.checkInterval = 10 * 60 * 1000; // 10 分钟检查一次
  }

    // 距离上次检查不足 10 分钟，跳过
    if (Date.now() - this.lastCheckTime < this.checkInterval) {
      return { quality: 'unknown', ip: 'cached' };

        return { quality: 'blacklisted', ip: currentIP };

      // 检查访问 BOSS 直聘的响应时间
      const startTime = Date.now();
      const bossResponse = await fetchWithTimeout('https://www.zhipin.com/', 10000);

        console.warn(`⚠️ IP ${currentIP} 响应慢 (${responseTime}ms)`);
        return { quality: 'slow', ip: currentIP };
      }
      
      // 检查响应中是否有异常提示
      if (bossResponse.text?.includes('访问异常') ||

      return { quality: 'good', ip: currentIP };

      console.error('IP 检查失败:', error.message);
      return { quality: 'unknown', ip: 'error' };
    }
  }
  
  // IP 熔断：连续 3 次检测到问题，切换代理
  async checkAndSwitchIP() {

    return true;
  }
}
```

## 三、预判能力强化方案

### 3.1 交互限制预判

#### ❌ 原错误：只搜索不限制

#### ✅ 修正方案：操作计数器 + 主动限流

```javascript
class OperationLimiter {
  constructor() {
    this.limits = {
      // 每小时最大搜索次数
      searchesPerHour: 10,
      
      // 每天最大职位查看数
      jobDetailsPerDay: 30,
      
      // 每小时最大点击次数
      clicksPerHour: 50,
      
      // 两次搜索最小间隔（分钟）
      minSearchInterval: 5,
      
      // 每天最大工作时长（小时）
      maxWorkHours: 4
    };
    
    this.counters = {
      searches: [],
      jobDetails: [],
      clicks: [],
      startTime: Date.now()
    };
    
    this.loadCounters();
  }
  
  loadCounters() {
    const file = 'operation-counters.json';
    try {
      if (fs.existsSync(file)) {
        this.counters = JSON.parse(fs.readFileSync(file, 'utf8'));
      }
    } catch (e) {
      console.warn('读取计数器失败');
    }
  }
  
  saveCounters() {
    fs.writeFileSync('operation-counters.json', JSON.stringify(this.counters));
  }
  
  // 清理过期记录（只保留最近的）
  pruneRecords() {
    const oneHourAgo = Date.now() - 60 * 60 * 1000;
    const oneDayAgo = Date.now() - 24 * 60 * 60 * 1000;
    
    this.counters.searches = this.counters.searches.filter(t => t > oneHourAgo);
    this.counters.clicks = this.counters.clicks.filter(t => t > oneHourAgo);
    this.counters.jobDetails = this.counters.jobDetails.filter(t => t > oneDayAgo);
    
    this.saveCounters();
  }
  
  // 检查是否可以执行搜索
  canSearch() {
    this.pruneRecords();
    
    // 检查每小时搜索次数
    if (this.counters.searches.length >= this.limits.searchesPerHour) {
      console.warn('⚠️ 已达到每小时搜索上限，等待下一小时');
      return false;
    }
    
    // 检查搜索间隔
    const lastSearch = this.counters.searches[this.counters.searches.length - 1];
    if (lastSearch && Date.now() - lastSearch < this.limits.minSearchInterval * 60 * 1000) {
      const waitTime = this.limits.minSearchInterval * 60 * 1000 - (Date.now() - lastSearch);
      console.warn(`⚠️ 搜索间隔太短，需等待 ${Math.round(waitTime/1000)} 秒`);
      return false;
    }
    
    // 检查每日工作时长
    const workHours = (Date.now() - this.counters.startTime) / (60 * 60 * 1000);
    if (workHours > this.limits.maxWorkHours) {
      console.warn('⚠️ 已达到每日最大工作时长，明日继续');
      return false;
    }
    
    return true;
  }
  
  // 记录搜索操作
  recordSearch() {
    this.counters.searches.push(Date.now());
    this.saveCounters();
  }
  
  // 记录职位详情查看
  recordJobDetail() {
    this.counters.jobDetails.push(Date.now());
    
    if (this.counters.jobDetails.length >= this.limits.jobDetailsPerDay) {
      console.warn('⚠️ 已达到每日职位查看上限');
    }
    
    this.saveCounters();
  }
}

// 使用
const limiter = new OperationLimiter();

async function searchWithLimit(keyword) {
  if (!limiter.canSearch()) {
    throw new Error('操作受限，等待限流解除');
  }
  
  limiter.recordSearch();
  await performSearch(keyword);
}
```

**核心改进：**
- 多维度限流（搜索次数、查看详情次数、点击次数）
- 时间窗口限流（每小时、每天）
- 最小间隔强制等待
- 每日工作时长限制
- 持久化计数器（Session 重置不丢失）

### 3.2 数据孤岛问题

#### ❌ 原错误：去重依赖本地缓存

#### ✅ 修正方案：外部数据库 + 唯一 ID

```javascript
class JobDatabase {
  constructor() {
    this.dbFile = 'jobs-db.json';
    this.db = this.loadDB();
  }
  
  loadDB() {
    try {
      if (fs.existsSync(this.dbFile)) {
        return JSON.parse(fs.readFileSync(this.dbFile, 'utf8'));
      }
    } catch (e) {
      console.warn('读取数据库失败');
    }
    return { jobs: {}, lastCleanup: Date.now() };
  }
  
  saveDB() {
    fs.writeFileSync(this.dbFile, JSON.stringify(this.db, null, 2));
  }
  
  // 生成职位唯一 ID（不依赖 URL）
  generateJobId(jobInfo) {
    const hash = crypto.createHash('md5');
    hash.update(`${jobInfo.company}-${jobInfo.title}-${jobInfo.salary}-${jobInfo.district}`);
    return hash.digest('hex');
  }
  
  // 检查职位是否已处理
  isJobProcessed(jobInfo) {
    const jobId = this.generateJobId(jobInfo);
    const record = this.db.jobs[jobId];
    
    if (!record) {
      return false;
    }
    
    // 30 天内的记录视为已处理
    const thirtyDaysAgo = Date.now() - 30 * 24 * 60 * 60 * 1000;
    if (record.processedAt > thirtyDaysAgo) {
      console.log(`✅ 职位已处理（${record.processedAt}）`);
      return true;
    }
    
    return false;
  }
  
  // 记录已处理职位
  markJobProcessed(jobInfo, status = 'viewed') {
    const jobId = this.generateJobId(jobInfo);
    this.db.jobs[jobId] = {
      jobInfo,
      status, // viewed, contacted, applied, interviewed, offered
      processedAt: Date.now(),
      notes: []
    };
    this.saveDB();
  }
  
  // 定期清理（每周）
  cleanup() {
    const oneWeekAgo = Date.now() - 7 * 24 * 60 * 60 * 1000;
    if (this.db.lastCleanup < oneWeekAgo) {
      const ninetyDaysAgo = Date.now() - 90 * 24 * 60 * 60 * 1000;
      
      // 删除 90 天前的记录
      for (const jobId in this.db.jobs) {
        if (this.db.jobs[jobId].processedAt < ninetyDaysAgo) {
          delete this.db.jobs[jobId];
        }
      }
      
      this.db.lastCleanup = Date.now();
      this.saveDB();
      console.log('🗑️ 数据库清理完成');
    }
  }
  
  // 获取统计信息
  getStats() {
    const today = new Date().toDateString();
    const todayJobs = Object.values(this.db.jobs).filter(
      r => new Date(r.processedAt).toDateString() === today
    );
    
    return {
      totalJobs: Object.keys(this.db.jobs).length,
      todayJobs: todayJobs.length,
      byStatus: {
        viewed: todayJobs.filter(r => r.status === 'viewed').length,
        contacted: todayJobs.filter(r => r.status === 'contacted').length,
        applied: todayJobs.filter(r => r.status === 'applied').length
      }
    };
  }
}
```

### 3.3 静态等待漏洞

#### ❌ 原错误：固定 15-20 秒死等

#### ✅ 修正方案：事件驱动 + 动态超时

```javascript
class SmartWaiter {
  constructor() {
    this.defaultTimeout = 30000;
    this.quickTimeout = 10000;
    this.longTimeout = 60000;
  }
  
  // 等待特定元素出现
  async waitForElement(selector, options = {}) {
    const timeout = options.timeout || this.defaultTimeout;
    const checkInterval = options.checkInterval || 3000;
    const startTime = Date.now();
    
    console.log(`⏳ 等待元素：${selector}`);
    
    while (Date.now() - startTime < timeout) {
      const snapshot = await browser.snapshot({ refs: 'aria', timeoutMs: 5000 });
      
      // 检查目标元素
      const element = snapshot.querySelector(selector);
      if (element) {
        console.log(`✅ 元素已就绪：${selector}`);
        return element;
      }
      
      // 检查是否有错误页面
      if (this.isErrorPage(snapshot)) {
        throw new Error('检测到错误页面，停止等待');
      }
      
      // 指数退避
      await sleep(checkInterval + Math.random() * 2000);
    }
    
    throw new Error(`等待超时：${selector}`);
  }
  
  // 等待页面进入特定状态
  async waitForState(stateChecker, options = {}) {
    const timeout = options.timeout || this.defaultTimeout;
    const startTime = Date.now();
    let lastState = null;
    let unchangedCount = 0;
    
    while (Date.now() - startTime < timeout) {
      const snapshot = await browser.snapshot({ refs: 'aria', timeoutMs: 5000 });
      const currentState = stateChecker(snapshot);
      
      // 状态达成
      if (currentState === 'ready') {
        console.log('✅ 页面状态就绪');
        return true;
      }
      
      // 检测错误状态
      if (currentState === 'error') {
        throw new Error('页面进入错误状态');
      }
      
      // 检测状态是否长时间不变
      if (currentState === lastState) {
        unchangedCount++;
        if (unchangedCount >= 3) {
          console.warn('⚠️ 页面状态长时间未变化，可能卡住');
        }
      } else {
        unchangedCount = 0;
      }
      
      lastState = currentState;
      await sleep(5000 + Math.random() * 3000);
    }
    
    throw new Error('状态等待超时');
  }
  
  // 检测错误页面
  isErrorPage(snapshot) {
    const errorKeywords = [
      '验证码', '安全验证', '访问异常',
      '请稍后再试', '网络错误', '403', '404'
    ];
    
    const text = snapshot.text?.toLowerCase() || '';
    return errorKeywords.some(keyword => text.includes(keyword.toLowerCase()));
  }
}

// 使用
const waiter = new SmartWaiter();

async function loadJobList() {
  // 等待职位列表出现
  await waiter.waitForElement('[ref*="job"]', { timeout: 30000 });
  
  // 或者等待特定状态
  await waiter.waitForState((snapshot) => {
    const hasJobs = snapshot.querySelectorAll('[ref*="job"]').length > 0;
    const hasError = snapshot.text?.includes('验证码');
    
    if (hasError) return 'error';
    if (hasJobs) return 'ready';
    return 'loading';
  });
}
```

**核心改进：**
- 等待特定元素而非固定时间
- 状态机检测（loading→ready/error）
- 错误页面快速失败
- 状态不变检测（防卡住）
- 动态超时（根据场景调整）
