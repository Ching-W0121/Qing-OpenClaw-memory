[51] 新增：

[53] 核心：
[54] class IndustryMatcher:

    def __init__(self, tree):
        self.tree = tree

    def match(self, user_industry, job_industry):

        if user_industry == job_industry:
            return 1.0

        # 子行业
        if job_industry in self.tree.get(user_industry, []):
            return 0.8

        # 同级行业
        for parent, childs in self.tree.items():
            if user_industry in childs and job_industry in childs:
                return 0.6

        return 0.2

    async def run(self, user_id):

        profile = ResumeService.get_profile(user_id)

        jobs = JobService.search(profile.target_jobs)
